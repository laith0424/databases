SELECT name ,  country.region , countrylanguage.language , countrylanguage.isofficial , ss.result  FROM 
    ( SELECT  region , language , isofficial , 
      IF ( COUNT( name ) > 1  , 'There are more than one country' , 'There is only one country' 
         ) AS result
           FROM country  
           JOIN countrylanguage ON countrylanguage.countrycode = code 
           WHERE  (region , language , isofficial ) = 
              ( SELECT region ,  language, isofficial   
                 FROM country INNER JOIN countrylanguage ON code = countrycode 
                 WHERE name = ? AND isofficial ='T'
               ) 
    ) AS ss
     JOIN country ON country.region = ss.region  JOIN countrylanguage ON countrylanguage.countrycode = code 
     WHERE  (country.region , countrylanguage.language , countrylanguage.isofficial ) =  ( ss.region ,  ss.language, ss.isofficial)
